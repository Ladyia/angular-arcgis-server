Example options settings

```javascript
options = {
    folder: 'GEWA',
    layer: 'Streams',
    service: 'gewa_sde',
    server: 'FeatureServer' || 'MapServer' || 'GPServer',
    actions: 'query',
    params: {},
    header: {
      'Content-Type': undefined
    },
    timeout: 5000
  };

```
